UTILITIES

Shadows

Add or remove shadows to elements with box-shadow utilities.

export const shadows = {
  none: 'none',
  dark: {
    sm: '0 2px 8px 0 rgba(0, 0, 0, 0.48), 0 1px 2px 0 rgba(0, 0, 0, 0.16)',
    md: '0 4px 16px 0 rgba(0, 0, 0, 0.48), 0 2px 4px 0 rgba(0, 0, 0, 0.16)',
    lg: '0 8px 32px 0 rgba(0, 0, 0, 0.48), 0 4px 8px 0 rgba(0, 0, 0, 0.16)'
  },
  light: {
    sm: '0 2px 8px 0 rgba(0, 0, 0, 0.16), 0 1px 2px 0 rgba(0, 0, 0, 0.08)',
    md: '0 4px 16px 0 rgba(0, 0, 0, 0.16), 0 2px 4px 0 rgba(0, 0, 0, 0.08)',
    lg: '0 8px 32px 0 rgba(0, 0, 0, 0.16), 0 4px 8px 0 rgba(0, 0, 0, 0.08)'
  }
}
boxShadow=sm
boxShadow=md
boxShadow=lg
EDITABLE EXAMPLE
function Example() {
  const { colorMode } = useColorMode();
  return (
    <Stack direction="row" spacing="6x" justifyContent="center">
      <Box width="56x" height="24x" p="2x" bg="gray:70" boxShadow={`${colorMode}.sm`}>boxShadow=sm</Box>
      <Box width="56x" height="24x" p="2x" bg="gray:70" boxShadow={`${colorMode}.md`}>boxShadow=md</Box>
      <Box width="56x" height="24x" p="2x" bg="gray:70" boxShadow={`${colorMode}.lg`}>boxShadow=lg</Box>
    </Stack>
  );
}
render(<Example />);
