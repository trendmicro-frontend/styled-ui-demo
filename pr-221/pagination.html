function Minor() {
  const [page, setPage] = React.useState(3);
  const totalPages = 20;
  return (
    <Flex>
      <Flex mr="3x" alignItems="center">
        <Text>
          Total: 400
        </Text>
        <Divider vertical height="6x" ml="3x" />
        <Menu>
          <MenuButton variant="ghost">10 per page</MenuButton>
          <MenuList>
            <MenuItem>10</MenuItem>
            <MenuItem>50</MenuItem>
            <MenuItem>100</MenuItem>
          </MenuList>
        </Menu>
        <Divider vertical height="6x" />
      </Flex>
      <Pagination count={totalPages} page={page} onChange={(_, value) => setPage(value)} />
    </Flex>
  );
}

function Minor() {
  const [page, setPage] = React.useState(1);
  const totalPages = 40;
  const prevPageDisabled = (page <= 1);
  const nextPageDisabled = (page >= totalPages);
  const onInputChange = (e) => {
    const value = e.target.value;
    if (value <= 1) {
      setPage(1);
    } else if (value >= totalPages) {
      setPage(totalPages);
    } else if (!isNaN(value)) {
      setPage(e.target.value);
    }
  }
  return (
    <Flex>
      <Flex mr="3x" alignItems="center">
        <Text>
          Total: 400
        </Text>
        <Divider vertical height="6x" ml="3x" />
        <Menu>
          <MenuButton variant="ghost">50 per page</MenuButton>
          <MenuList>
            <MenuItem>10</MenuItem>
            <MenuItem>50</MenuItem>
            <MenuItem>100</MenuItem>
          </MenuList>
        </Menu>
        <Input width={32} px={0} textAlign="center" onChange={onInputChange} value={page} />
        <Space width="2x" />
        <Text>/</Text>
        <Space width="2x" />
        <Text>{totalPages}</Text>
      </Flex>
      <ButtonGroup variant="secondary">
        <Button
          disabled={prevPageDisabled}
          onClick={(event) => {
            const prevPage = page > 1 ? page - 1 : page;
            if (prevPage !== page) {
              setPage(prevPage);
            }
          }}
        >
          <Icon icon="angle-left" />
        </Button>
        <Button
          disabled={nextPageDisabled}
          onClick={(event) => {
            const nextPage = page < totalPages ? page + 1 : page;
            if (nextPage !== page) {
              setPage(nextPage);
            }
          }}
        >
          <Icon icon="angle-right" />
        </Button>
      </ButtonGroup>
    </Flex>
  );
}

Props

Pagination composes the ghost Button component. You can override the default styles with style props.

NameTypeDefaultDescription
countnumber1The total number of pages.
disabledbooleanfalseIf true, the pagination component will be disabled.
defaultPagenumber1The page selected by default when the component is uncontrolled.
pagenumberThe current page. when the component is controlled.
prevLabelstring or react element'<'The previous button label
nextLabelstring or react element'>'The next button label
ellipsisLabelstring or react element'...'The next button label
boundaryCountnumber1Number of always visible pages at the beginning and end.
siblingCountnumber1Number of always visible pages before and after the current page.
onChangefunctionCallback fired when the page is changed.

Signature:
function(event: object, page: number) => void
event: The event source of the callback.
page: The page selected.
